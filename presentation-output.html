Why Sass?

Mixins: Setup


@mixin breakpoint($point) {
  @if $point == designer {
    @media (max-width: 1600px)
    { @content; }
  }

  @if $point == developer {
    @media (max-width: 1250px)
    { @content; }
  }
}



.breakpoint-example {

  @include breakpoint(designer) {
    color:red;
  }

  @include breakpoint(developer){
    color:blue;
  }
 }
}
